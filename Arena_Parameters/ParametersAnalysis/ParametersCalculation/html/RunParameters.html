function RunParameters(~,~)

%--------------------To calculate all the parameters by using all
%GUI--------

global h

Variables

%-------------------Select the experiments-------------------
index_selected = get(h.listRunParams,'Value');
list = get(h.listRunParams,'String');
item_selected = list(index_selected,1); % Convert from cell array

Exp_arrA=item_selected'; %Input Data

-----------------------Select the dates of the mat file -----------------------

for j=1:length(item_selected)
Folder_Data=strcat(get(h.editRunParams,'string'),'\',item_selected(j),'\','Results','\');
Folder_Data=char(Folder_Data);
ListFiles=dir([Folder_Data,'*.mat']);

clear ListDates
for i=3:length(ListFiles(not([ListFiles.isdir])))
I=strfind(ListFiles(i).name,'_');
Aux=char(ListFiles(i).name);
ListDates(i-2)=cellstr(Aux(1:I(3)-1));
end
subStr_arrA{1,j}=unique(ListDates);

end

aux_data.mf_nmb=0;
aux_data.excel_file='AllParametersGV.xlsx';

for w=1:length(item_selected)

[prmsGV37_data,T,Y,non_empty_days,aux_data]=func_extractPrms_byAllParameters15_1_14(Exp_arr,subStr_arr,aux_data);

 aa=prmsGV37_data{1}(:,:,logical(non_empty_days)); prmsGV37_data{1}=aa;

                prms.time_interval_mode=0; prms.time_interval_mode=0; %length(subStr_arr);
                prms.Exp=Exp_arr;
                prms.rootPath=aux_data.rootPath;
                prms.mm_nmb=5;
                prms.mf_nmb=0;
                prms.excel_file_path=strcat(aux_data.save_to,aux_data.excel_file);
                prms.write2excel_Mode=1

                [Ok,data4excel]=writeGV372excel(prmsGV37_data,T,Y,prms,1,data4excel,subStr_arr,Exp_arr);
                writing2excel_done=1;data4mat(1).prmsGV37_data=prmsGV37_data{1}; data4mat(1).T=T; data4mat(1).Y=Y; data4mat(1).non_empty_days=non_empty_days;

end

end

function [Ok data4excel]=writeGV372excel(prmsGV37_data,T,Y,prms,visitN,data4excel,subStr_arr,Exp_arr)
% write to excel_file
 dd=prmsGV37_data;

Ok=0;

if(nargin<6), data4excel=[];  end
tt=data4excel;
if(nargin<5 || isempty(visitN)), visitN=1; end

IDs_clmn = get_mIDs_by_ExpName(prms.Exp,prms.rootPath)';

mNmb=prms.mm_nmb+prms.mf_nmb;
sz=size(prmsGV37_data{1})
% assert(length(T)==sz(2),'Not the same number of Titles and Measurements !?');
% assert(sz(1)==mNmb,'Problem with data size');
if(prms.time_interval_mode>0)
    timeIntrvlsNmb=prms.time_interval_mode;
%     if(~isempty(headers)), headers{end+1}=prms.time_interval;
%     else
      headers=cat(2,{' '},{'head chip'},{prms.time_interval});
%     end
else
    timeIntrvlsNmb=sz(3);
    tn=mat2cell(1:timeIntrvlsNmb,1,ones(1,timeIntrvlsNmb)); headers=cat(2,{' '},{'head chip'},tn);
end

excel_file=prms.excel_file_path;

firstOffst=3; gapping=1; % gapping - number of empty lines between 2 data chunks
% currentStep=mNmb+gapping;  % is defined as number of lines between first line of previous data(including title) and first line of the next one
% currentOffst=firstOffst;

head_ref=['A' num2str(firstOffst-1)];

% xlswrite(excel_file, headers, visitN, head_ref);

if(~isempty(tt)), data4excel{1,end+1}=prms.time_interval;
else
    data4excel=headers;
end
[m n p]=size(cell2mat(dd))
for dn=1:n
    if(isempty(tt))
        cr_data=cell(mNmb,1); cr_data{1}=[T{dn} '(' Y{dn} ')']; % <-- first cell column with current parameter name
        cr_data=cat(2,cr_data,IDs_clmn); % <-- IDs column
    end
    if(prms.time_interval_mode>0)
        aa=mat2cell(squeeze(prmsGV37_data{1}(:,dn)),ones(1,sz(1)),ones(1,timeIntrvlsNmb));
    else
        aa=mat2cell(squeeze(prmsGV37_data{1}(:,dn,:)),ones(1,sz(1)),ones(1,timeIntrvlsNmb));
    end
    if(isempty(tt))
        cr_data=cat(2,cr_data,aa); % horizontal addition of params data
        cr_data=cat(1,cr_data,cell(gapping,size(headers,2)));
        data4excel=cat(1,data4excel,cr_data);
    else
        assert(mNmb==length(aa),'Problem with mNmb');
        for tk=1:mNmb
            data4excel{tk+1+(dn-1)*(1+mNmb),end}=aa{tk};
        end
    end
end

% dd1=permute(dd{:,:,:},[2 1 3]);
% dd2=reshape(dd1,[n,m*p]);
% A=strcat(T,'(',Y,')');
% ident=IDs_clmn';
% subStr_arr{1};
% raw(3:length(A)+2,1)=A';
% raw(2,3:p*m+2)=repmat(ident,[1,p]);
% raw(2,1)={'Head chip'};
% raw(3:size(dd2,1)+2,2:size(dd2,2)+1)=num2cell(dd2);
% raw(1,1)={'Dates'};
% raw(1,3:length(ident):length(subStr_arr{1})+2)=subStr_arr{1};

sheet=Exp_arr{1};
% xlswrite(excel_file, data4excel, visitN, head_ref);
% xlrange=[visitN,head_ref];
xlswrite(excel_file, data4excel,sheet);
% xlswrite(excel_file,raw,sheet);
Ok=1;

end
